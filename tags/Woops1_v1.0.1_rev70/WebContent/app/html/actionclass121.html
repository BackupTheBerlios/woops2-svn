035 /**
036  * This class is responsible to instantiate our ListControl.
037  * This class also implements the methods which are called if
038  * an event in our ListControl occurs.
039  *
040  * @author    <a href="mailto:gschulz@scc-gmbh.com">Gernot Schulz</a>
041  * @version  $Revision: 1.1 $
042  */
043 public class AddressBrowseAction extends FWAction {
044 
045   /**
046    * Constructor for AddressBrowseAction.
047    */
048   public AddressBrowseAction() {
049     super();
050   }
051 
052   /**
053    * @see com.cc.framework.adapter.struts.FWAction#doExecute(ActionContext)
054    */
055   public void doExecute(ActionContext ctx) throws IOException, ServletException {
056 
057     try {
058       this.refreshList(ctx);
059     } catch (Throwable t) {
060       log.error(t);
061       ctx.addGlobalError(Messages.ERROR_INIT_FORMBEAN, t.getClass().getName() + " " + t.getMessage());
062     }
063     
064     // Display the Page with the List
065     ctx.forwardToInput();
066   }
067   
068   
069   /**
070    * Fill's and refresh's the ListControl from the Database
071    * @param  ctx  ControlActionContext
072    * @throws  Exception
073    */
074   private void refreshList(ActionContext ctx) throws Exception {
075 
076     SimpleListControl addressList = null;
077     
078     // first get the Displaydata for our List
079     AddressDisplayList dspData = DBAddress.fetch();
080 
081     if (null == ctx.session().getAttribute("addresses")) {
082       // secondly create the ListControl
083       addressList = new SimpleListControl();
084 
085       // Now we set the DesignModel for the Control
086       addressList.setDesignModel(doCreateDesignModel());
087     
088       ctx.session().setAttribute("addresses", addressList);  
089     } else {
090       addressList = (SimpleListControl) ctx.session().getAttribute("addresses");
091     }
092     
093     // Populate the Control with the Data to display
094     addressList.setDataModel(dspData);
095   }
096   
097 
098   /**
099    * Generates a DesignModel for the ListControl
100    * 
101    * @return  ListDesignModel
102    * @throws  Exception
103    */
104   private ListDesignModel doCreateDesignModel() throws Exception {
105 
106     ListDesignModel ldm = new ListDesignModelImp();
107     ldm.setTitle("Dynamic AddressList");
108     ldm.setRowCount(12);
109     ldm.setEmptyText("No Addresses found!");
110     ldm.setAction("sample121/addressBrowse");
111     ldm.setName("addresses");
112     ldm.setId("addlist1");
113 
114 
115     // =======================================
116     // Add a DrilldownColumn
117     // =======================================
118     ColumnDrilldownDesignModelImp dcdm = new ColumnDrilldownDesignModelImp();
119     
120     dcdm.setTitle("ZipCode");
121     dcdm.setWidth(Integer.toString(50));
122     dcdm.setProperty("zipcode");
123     dcdm.setHandler(ClientEvent.ONCLICK, "javascript:alert('Not implemented in this Sample'); return false;");
124     
125     ldm.addColumn(dcdm);
126 
127     // =======================================
128     // Add a TextColumn
129     // =======================================
130     ColumnTextDesignModelImp tcdm = new ColumnTextDesignModelImp();
131 
132     tcdm.setTitle("City");
133     tcdm.setWidth(Integer.toString(150));
134     tcdm.setProperty("city");
135     
136     ldm.addColumn(tcdm);
137 
138     // =======================================
139     // Add a TextColumn
140     // =======================================
141     tcdm = new ColumnTextDesignModelImp();
142 
143     tcdm.setTitle("Street");
144     tcdm.setWidth(Integer.toString(220));
145     tcdm.setProperty("street");
146     
147     ldm.addColumn(tcdm);
148 
149     // =======================================
150     // Add a EditColumn
151     // =======================================
152     ColumnEditDesignModelImp ecdm = new ColumnEditDesignModelImp();
153     ecdm.setTitle("Edit");
154     ecdm.setHandler(ClientEvent.ONCLICK, "alert('Not implemented in this Sample'); return false;");
155         
156     ldm.addColumn(ecdm);
157 
158     // =======================================
159     // Add a DeleteColumn
160     // =======================================
161     ColumnDeleteDesignModelImp decdm = new ColumnDeleteDesignModelImp();
162     decdm.setTitle("Delete");
163     decdm.setHandler(ClientEvent.ONCLICK, "alert('Not implemented in this Sample'); return false;");
164     
165     ldm.addColumn(decdm);
166     
167     return ldm;
168   }
